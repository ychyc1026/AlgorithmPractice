## 二、栈的主流实现（二叉树非递归算法依赖）

栈用于二叉树非递归遍历、非递归交换左右子树，实现两种主流版本：顺序栈（数组实现，高效）、链栈（链表实现，灵活，无容量限制）。

## 2\.1 顺序栈实现（存储二叉树节点指针）

```c
#define STACK_SIZE 100 // 顺序栈容量，可根据二叉树深度调整

// 顺序栈结构
typedef struct {
    BiTree data[STACK_SIZE]; // 存储二叉树节点指针
    int top;                 // 栈顶指针：top=-1（空栈），top=STACK_SIZE-1（满栈）
} SqStack;

// 1. 初始化栈
void InitSqStack(SqStack *S) {
    S->top = -1; // 空栈标识
}

// 2. 判断栈是否为空
bool SqStackEmpty(SqStack *S) {
    return S->top == -1;
}

// 3. 判断栈是否满
bool SqStackFull(SqStack *S) {
    return S->top == STACK_SIZE - 1;
}

// 4. 入栈（压入二叉树节点指针）
bool PushSqStack(SqStack *S, BiTree node) {
    if (SqStackFull(S)) {
        printf("顺序栈满，入栈失败！\n");
        return false;
    }
    S->data[++S->top] = node; // 先移动栈顶，再存入数据
    return true;
}

// 5. 出栈（弹出栈顶节点指针）
bool PopSqStack(SqStack *S, BiTree *node) {
    if (SqStackEmpty(S)) {
        printf("顺序栈空，出栈失败！\n");
        return false;
    }
    *node = S->data[S->top--]; // 先取出数据，再移动栈顶
    return true;
}

// 6. 获取栈顶元素（不弹出）
bool GetTopSqStack(SqStack *S, BiTree *node) {
    if (SqStackEmpty(S)) {
        printf("顺序栈空，无栈顶元素！\n");
        return false;
    }
    *node = S->data[S->top];
    return true;
}

// 7. 销毁顺序栈（数组实现，无需手动释放内存，仅重置栈顶）
void DestroySqStack(SqStack *S) {
    S->top = -1;
    printf("顺序栈销毁完成！\n");
}

```

## 2\.2 链栈实现（存储二叉树节点指针）

```c
// 链栈节点结构
typedef struct StackNode {
    BiTree data;                // 存储二叉树节点指针
    struct StackNode *next;     // 指向下一个栈节点
} StackNode, *LinkStack;

// 1. 初始化链栈（栈顶指针为NULL，即为空栈）
void InitLinkStack(LinkStack *top) {
    *top = NULL;
}

// 2. 判断链栈是否为空
bool LinkStackEmpty(LinkStack top) {
    return top == NULL;
}

// 3. 入栈（头插法，效率高）
bool PushLinkStack(LinkStack *top, BiTree node) {
    // 新建栈节点
    StackNode *newNode = (StackNode *)malloc(sizeof(StackNode));
    if (newNode == NULL) {
        printf("内存分配失败，入栈失败！\n");
        return false;
    }
    newNode->data = node;
    newNode->next = *top; // 新节点指向原栈顶
    *top = newNode;       // 栈顶指针指向新节点
    return true;
}

// 4. 出栈（头删法）
bool PopLinkStack(LinkStack *top, BiTree *node) {
    if (LinkStackEmpty(*top)) {
        printf("链栈空，出栈失败！\n");
        return false;
    }
    StackNode *temp = *top;     // 暂存栈顶节点
    *node = temp->data;         // 取出栈顶数据
    *top = temp->next;          // 栈顶指针下移
    free(temp);                 // 释放原栈顶节点内存
    temp = NULL;
    return true;
}

// 5. 获取栈顶元素（不弹出）
bool GetTopLinkStack(LinkStack top, BiTree *node) {
    if (LinkStackEmpty(top)) {
        printf("链栈空，无栈顶元素！\n");
        return false;
    }
    *node = top->data;
    return true;
}

// 6. 销毁链栈（遍历释放所有节点）
void DestroyLinkStack(LinkStack *top) {
    StackNode *temp;
    while (*top != NULL) {
        temp = *top;
        *top = (*top)->next;
        free(temp);
    }
    temp = NULL;
    printf("链栈销毁完成！\n");
}

```

## 三、队列的主流实现（二叉树层次遍历依赖）

队列用于二叉树层次遍历，实现两种主流版本：顺序队列（数组实现）、循环队列（数组实现，解决顺序队列假溢出问题）、链队列（链表实现，无容量限制）。

## 3\.1 顺序队列实现（存储二叉树节点指针）

```c
#define QUEUE_SIZE 100 // 顺序队列容量，可根据二叉树宽度调整

// 顺序队列结构
typedef struct {
    BiTree data[QUEUE_SIZE]; // 存储二叉树节点指针
    int front;               // 队头指针（指向队头元素）
    int rear;                // 队尾指针（指向队尾元素的下一个空位）
} SqQueue;

// 1. 初始化顺序队列
void InitSqQueue(SqQueue *Q) {
    Q->front = Q->rear = 0; // 空队列标识：front == rear
}

// 2. 判断顺序队列是否为空
bool SqQueueEmpty(SqQueue *Q) {
    return Q->front == Q->rear;
}

// 3. 判断顺序队列是否满
bool SqQueueFull(SqQueue *Q) {
    return Q->rear == QUEUE_SIZE; // 存在假溢出问题
}

// 4. 入队（队尾插入）
bool EnSqQueue(SqQueue *Q, BiTree node) {
    if (SqQueueFull(Q)) {
        printf("顺序队列满，入队失败！\n");
        return false;
    }
    Q->data[Q->rear++] = node; // 存入数据，队尾指针后移
    return true;
}

// 5. 出队（队头删除）
bool DeSqQueue(SqQueue *Q, BiTree *node) {
    if (SqQueueEmpty(Q)) {
        printf("顺序队列为空，出队失败！\n");
        return false;
    }
    *node = Q->data[Q->front++]; // 取出队头数据，队头指针后移
    return true;
}

// 6. 获取队头元素（不弹出）
bool GetHeadSqQueue(SqQueue *Q, BiTree *node) {
    if (SqQueueEmpty(Q)) {
        printf("顺序队列为空，无队头元素！\n");
        return false;
    }
    *node = Q->data[Q->front];
    return true;
}

// 7. 销毁顺序队列（重置队头队尾即可）
void DestroySqQueue(SqQueue *Q) {
    Q->front = Q->rear = 0;
    printf("顺序队列销毁完成！\n");
}

```

## 3\.2 循环队列实现（推荐，解决假溢出）

```c
#define CIRCLE_QUEUE_SIZE 100 // 循环队列容量

// 循环队列结构
typedef struct {
    BiTree data[CIRCLE_QUEUE_SIZE];
    int front; // 队头指针（指向队头元素）
    int rear;  // 队尾指针（指向队尾元素的下一个空位）
} CircleQueue;

// 1. 初始化循环队列
void InitCircleQueue(CircleQueue *Q) {
    Q->front = Q->rear = 0;
}

// 2. 判断循环队列是否为空
bool CircleQueueEmpty(CircleQueue *Q) {
    return Q->front == Q->rear;
}

// 3. 判断循环队列是否满（牺牲一个空位，避免与空队列混淆）
bool CircleQueueFull(CircleQueue *Q) {
    return (Q->rear + 1) % CIRCLE_QUEUE_SIZE == Q->front;
}

// 4. 入队
bool EnCircleQueue(CircleQueue *Q, BiTree node) {
    if (CircleQueueFull(Q)) {
        printf("循环队列满，入队失败！\n");
        return false;
    }
    Q->data[Q->rear] = node;
    Q->rear = (Q->rear + 1) % CIRCLE_QUEUE_SIZE; // 队尾指针循环后移
    return true;
}

// 5. 出队
bool DeCircleQueue(CircleQueue *Q, BiTree *node) {
    if (CircleQueueEmpty(Q)) {
        printf("循环队列为空，出队失败！\n");
        return false;
    }
    *node = Q->data[Q->front];
    Q->front = (Q->front + 1) % CIRCLE_QUEUE_SIZE; // 队头指针循环后移
    return true;
}

// 6. 获取队头元素
bool GetHeadCircleQueue(CircleQueue *Q, BiTree *node) {
    if (CircleQueueEmpty(Q)) {
        printf("循环队列为空，无队头元素！\n");
        return false;
    }
    *node = Q->data[Q->front];
    return true;
}

// 7. 销毁循环队列
void DestroyCircleQueue(CircleQueue *Q) {
    Q->front = Q->rear = 0;
    printf("循环队列销毁完成！\n");
}

```

## 3\.3 链队列实现（存储二叉树节点指针）

```c
// 链队列节点结构（队列元素节点）
typedef struct QueueNode {
    BiTree data;                // 存储二叉树节点指针
    struct QueueNode *next;     // 指向下一个队列节点
} QueueNode;

// 链队列结构（包含队头、队尾指针）
typedef struct {
    QueueNode *front; // 队头指针（指向队头节点）
    QueueNode *rear;  // 队尾指针（指向队尾节点）
} LinkQueue;

// 1. 初始化链队列（队头、队尾均为NULL，为空队列）
void InitLinkQueue(LinkQueue *Q) {
    Q->front = Q->rear = NULL;
}

// 2. 判断链队列是否为空
bool LinkQueueEmpty(LinkQueue *Q) {
    return Q->front == NULL;
}

// 3. 入队（队尾插入）
bool EnLinkQueue(LinkQueue *Q, BiTree node) {
    // 新建队列节点
    QueueNode *newNode = (QueueNode *)malloc(sizeof(QueueNode));
    if (newNode == NULL) {
        printf("内存分配失败，入队失败！\n");
        return false;
    }
    newNode->data = node;
    newNode->next = NULL;

    if (LinkQueueEmpty(Q)) {
        // 空队列时，队头、队尾均指向新节点
        Q->front = Q->rear = newNode;
    } else {
        // 非空队列，队尾节点指向新节点，队尾指针后移
        Q->rear->next = newNode;
        Q->rear = newNode;
    }
    return true;
}

// 4. 出队（队头删除）
bool DeLinkQueue(LinkQueue *Q, BiTree *node) {
    if (LinkQueueEmpty(Q)) {
        printf("链队列为空，出队失败！\n");
        return false;
    }
    QueueNode *temp = Q->front; // 暂存队头节点
    *node = temp->data;         // 取出队头数据

    if (Q->front == Q->rear) {
        // 只有一个节点时，出队后队列空，队头、队尾置为NULL
        Q->front = Q->rear = NULL;
    } else {
        // 多个节点时，队头指针后移
        Q->front = Q->front->next;
    }

    free(temp); // 释放原队头节点内存
    temp = NULL;
    return true;
}

// 5. 获取队头元素（不弹出）
bool GetHeadLinkQueue(LinkQueue *Q, BiTree *node) {
    if (LinkQueueEmpty(Q)) {
        printf("链队列为空，无队头元素！\n");
        return false;
    }
    *node = Q->front->data;
    return true;
}

// 6. 销毁链队列（遍历释放所有节点）
void DestroyLinkQueue(LinkQueue *Q) {
    QueueNode *temp;
    while (Q->front != NULL) {
        temp = Q->front;
        Q->front = Q->front->next;
        free(temp);
    }
    Q->rear = NULL; // 队尾指针置空
    temp = NULL;
    printf("链队列销毁完成！\n");
}

```

## 四、二叉树核心算法实现

所有算法均基于二叉链表实现，部分算法提供递归\+非递归版本，非递归版本默认使用顺序栈/循环队列（高效、易实现），可根据需求替换为链式版本。

## 4\.1 二叉树高度计算（递归\+非递归）

二叉树高度（深度）：从根节点到最远叶子节点的最长路径上的节点数（根节点高度为1）。

### 4\.1\.1 递归实现（简洁，易理解）

```c
// 递归计算二叉树高度
int BiTreeHeight_Recursive(BiTree T) {
    if (T == NULL) {
        return 0; // 空树高度为0
    }
    // 递归计算左子树高度
    int leftHeight = BiTreeHeight_Recursive(T->lchild);
    // 递归计算右子树高度
    int rightHeight = BiTreeHeight_Recursive(T->rchild);
    // 树的高度 = 左右子树中最高的高度 + 1（当前根节点）
    return (leftHeight > rightHeight ? leftHeight : rightHeight) + 1;
}

```

### 4\.1\.2 非递归实现（栈实现，后序遍历思想）

```c
// 非递归计算二叉树高度（顺序栈实现）
int BiTreeHeight_NonRecursive(BiTree T) {
    if (T == NULL) {
        return 0;
    }
    SqStack S;
    InitSqStack(&S);
    PushSqStack(&S, T); // 根节点入栈

    int maxHeight = 0;   // 记录最大高度（树的高度）
    int currHeight = 0;  // 记录当前节点所在高度

    // 辅助结构体：存储节点及其对应的高度
    typedef struct {
        BiTree node;
        int height;
    } StackElem;
    StackElem stack[STACK_SIZE];
    int top = -1;

    stack[++top].node = T;
    stack[top].height = 1;

    while (top != -1) {
        StackElem curr = stack[top--];
        currHeight = curr.height;

        // 更新最大高度
        if (currHeight > maxHeight) {
            maxHeight = currHeight;
        }

        // 右孩子先入栈（栈后进先出，保证左孩子先处理）
        if (curr.node->rchild != NULL) {
            stack[++top].node = curr.node->rchild;
            stack[top].height = currHeight + 1;
        }
        // 左孩子入栈
        if (curr.node->lchild != NULL) {
            stack[++top].node = curr.node->lchild;
            stack[top].height = currHeight + 1;
        }
    }

    DestroySqStack(&S);
    return maxHeight;
}

```

## 4\.2 二叉树总节点个数统计（递归\+非递归）

### 4\.2\.1 递归实现

```c
// 递归统计二叉树总节点个数
int BiTreeNodeCount_Recursive(BiTree T) {
    if (T == NULL) {
        return 0; // 空树节点数为0
    }
    // 总节点数 = 左子树节点数 + 右子树节点数 + 1（当前根节点）
    return BiTreeNodeCount_Recursive(T->lchild) + BiTreeNodeCount_Recursive(T->rchild) + 1;
}

```

### 4\.2\.2 非递归实现（栈实现，前序遍历思想）

```c
// 非递归统计二叉树总节点个数（顺序栈实现）
int BiTreeNodeCount_NonRecursive(BiTree T) {
    if (T == NULL) {
        return 0;
    }
    SqStack S;
    InitSqStack(&S);
    PushSqStack(&S, T);

    int count = 0; // 节点计数器

    while (!SqStackEmpty(&S)) {
        BiTree curr;
        PopSqStack(&S, &curr);
        count++; // 访问节点，计数器加1

        // 右孩子先入栈，左孩子后入栈（保证左孩子先访问）
        if (curr->rchild != NULL) {
            PushSqStack(&S, curr->rchild);
        }
        if (curr->lchild != NULL) {
            PushSqStack(&S, curr->lchild);
        }
    }

    DestroySqStack(&S);
    return count;
}

```

## 4\.3 二叉树叶子节点个数统计（递归\+非递归）

叶子节点：左右子树均为空的节点（lchild == NULL \&amp;\&amp; rchild == NULL）。

### 4\.3\.1 递归实现

```c
// 递归统计二叉树叶子节点个数
int BiTreeLeafCount_Recursive(BiTree T) {
    if (T == NULL) {
        return 0; // 空树无叶子节点
    }
    // 当前节点是叶子节点（左右子树均空）
    if (T->lchild == NULL && T->rchild == NULL) {
        return 1;
    }
    // 叶子节点总数 = 左子树叶子数 + 右子树叶子数
    return BiTreeLeafCount_Recursive(T->lchild) + BiTreeLeafCount_Recursive(T->rchild);
}

```

### 4\.3\.2 非递归实现（栈实现）

```c
// 非递归统计二叉树叶子节点个数（顺序栈实现）
int BiTreeLeafCount_NonRecursive(BiTree T) {
    if (T == NULL) {
        return 0;
    }
    SqStack S;
    InitSqStack(&S);
    PushSqStack(&S, T);

    int leafCount = 0; // 叶子节点计数器

    while (!SqStackEmpty(&S)) {
        BiTree curr;
        PopSqStack(&S, &curr);

        // 判断当前节点是否为叶子节点
        if (curr->lchild == NULL && curr->rchild == NULL) {
            leafCount++;
        }

        // 右孩子先入栈，左孩子后入栈
        if (curr->rchild != NULL) {
            PushSqStack(&S, curr->rchild);
        }
        if (curr->lchild != NULL) {
            PushSqStack(&S, curr->lchild);
        }
    }

    DestroySqStack(&S);
    return leafCount;
}

```

## 4\.4 二叉树层次遍历（队列实现）

层次遍历：从上到下、从左到右，依次访问每一层的节点（依赖队列，先进先出），此处用循环队列实现（推荐）。

```c
// 二叉树层次遍历（循环队列实现）
void LevelOrderTraverse(BiTree T, void(*Visit)(TElemType)) {
    if (T == NULL) {
        printf("二叉树为空，无法遍历！\n");
        return;
    }
    CircleQueue Q;
    InitCircleQueue(&Q);
    EnCircleQueue(&Q, T); // 根节点入队

    while (!CircleQueueEmpty(&Q)) {
        BiTree curr;
        DeCircleQueue(&Q, &curr); // 出队，访问当前节点
        Visit(curr->data);

        // 左孩子入队（先左后右，保证层次遍历顺序）
        if (curr->lchild != NULL) {
            EnCircleQueue(&Q, curr->lchild);
        }
        // 右孩子入队
        if (curr->rchild != NULL) {
            EnCircleQueue(&Q, curr->rchild);
        }
    }

    DestroyCircleQueue(&Q);
}

```

## 4\.5 递归实现二叉树左右子树交换

核心：递归交换每个节点的左右子树，自上而下（先交换根节点，再递归交换左右子树）。

```c
// 递归实现二叉树左右子树交换
void ExchangeBiTree_Recursive(BiTree T) {
    if (T == NULL) {
        return; // 空树无需交换
    }
    // 1. 交换当前节点的左右子树
    BiTree temp = T->lchild;
    T->lchild = T->rchild;
    T->rchild = temp;

    // 2. 递归交换左子树的左右子树
    ExchangeBiTree_Recursive(T->lchild);
    // 3. 递归交换右子树的左右子树
    ExchangeBiTree_Recursive(T->rchild);
}

```

## 4\.6 非递归实现二叉树左右子树交换（栈实现）

核心：用栈遍历所有节点，遍历到每个节点时，交换其左右子树（前序遍历思想）。

```c
// 非递归实现二叉树左右子树交换（顺序栈实现）
void ExchangeBiTree_NonRecursive(BiTree T) {
    if (T == NULL) {
        printf("二叉树为空，无需交换！\n");
        return;
    }
    SqStack S;
    InitSqStack(&S);
    PushSqStack(&S, T); // 根节点入栈

    while (!SqStackEmpty(&S)) {
        BiTree curr;
        PopSqStack(&S, &curr);

        // 交换当前节点的左右子树
        BiTree temp = curr->lchild;
        curr->lchild = curr->rchild;
        curr->rchild = temp;

        // 右孩子先入栈，左孩子后入栈（栈后进先出，保证左子树先交换）
        if (curr->rchild != NULL) {
            PushSqStack(&S, curr->rchild);
        }
        if (curr->lchild != NULL) {
            PushSqStack(&S, curr->lchild);
        }
    }

    DestroySqStack(&S);
}

```

## 4\.7 非递归实现二叉树中序遍历（栈实现）

中序遍历规则：左 → 根 → 右，用栈记录遍历路径，保证左子树先访问。

```c
// 非递归实现二叉树中序遍历（顺序栈实现）
void InOrderTraverse_NonRecursive(BiTree T, void(*Visit)(TElemType)) {
    if (T == NULL) {
        printf("二叉树为空，无法遍历！\n");
        return;
    }
    SqStack S;
    InitSqStack(&S);
    BiTree curr = T; // 工作指针，从根节点开始

    // 循环条件：工作指针不为空 或 栈不为空
    while (curr != NULL || !SqStackEmpty(&S)) {
        // 1. 一路向左走，所有左孩子入栈（直到左子树为空）
        while (curr != NULL) {
            PushSqStack(&S, curr);
            curr = curr->lchild;
        }

        // 2. 出栈，访问当前节点（根节点）
        PopSqStack(&S, &curr);
        Visit(curr->data);

        // 3. 转向右子树，重复上述过程
        curr = curr->rchild;
    }

    DestroySqStack(&S);
}

```

## 5\.1 代码使用说明

## 5\.2 常见问题解决

- 编译报错“invalid conversion”：大概率是Visit函数参数错误，需传入节点的data（如curr\-\&gt;data），而非节点本身。

- 栈溢出/队列溢出：增大栈/队列容量，或检查二叉树是否存在异常（如无限递归创建节点）。

- 遍历结果异常：检查栈/队列的入队/入栈顺序，尤其是非递归算法中“先右后左”的逻辑（栈后进先出）。

### 补充：二叉树递归创建函数（用于测试）

```c
// 递归创建二叉树（按前序遍历输入，#表示空节点）
void CreateBiTree(BiTree *T) {
    TElemType ch;
    // 注意：输入时需按前序顺序，空节点用#表示（如：AB#D##C##）
    scanf(" %c", &ch); // 空格避免读取换行符
    if (ch == '#') {
        *T = NULL;
    } else {
        // 分配节点内存
        *T = (BiTree)malloc(sizeof(BiTNode));
        if (*T == NULL) {
            printf("内存分配失败！\n");
            exit(1);
        }
        (*T)->data = ch;          // 赋值当前节点数据
        CreateBiTree(&((*T)->lchild)); // 递归创建左子树
        CreateBiTree(&((*T)->rchild)); // 递归创建右子树
    }
}

// 二叉树销毁（递归释放所有节点内存）
void DestroyBiTree(BiTree *T) {
    if (*T == NULL) {
        return;
    }
    // 先销毁左子树，再销毁右子树，最后销毁当前节点
    DestroyBiTree(&((*T)->lchild));
    DestroyBiTree(&((*T)->rchild));
    free(*T);
    *T = NULL; // 避免野指针
}

```
